pragma solidity 0.8.24;

contract TestContract {
    uint256 immutable init;
    address payable immutable _owner;
    constructor() payable {
        init = msg.value;
        _owner = payable(msg.sender);
    }
    function send() external  {
        _owner.transfer(init);
    }

}
